	python scripts/train.py --config configs/training_config.yaml

	pytest tests/ -v

	ruff check src/
	mypy src/

	ruff format src/

	rm -rf __pycache__
	rm -rf .pytest_cache
	rm -rf dist/
	rm -rf *.egg-info

	docker build -t my-ml-project .
